/**
 * Simple GUI demo

import namespace System:::Windows:::Input;

include("slutils.txt");

global $CANVAS;
global $cv;
$cv = new CanvasWrapper($CANVAS);

function OnClick()
{
	global $cv;
	$cv->moveCar->Begin();
	$cv->moveWheels->Begin();
}

function OnLoad()
{
	global $cv;
	$cv->carCanvas->MouseLeftButtonDown->Add
		(new MouseEventHandler("OnClick"));
}
